class Solution {
    public int romanToInt(String s) {
        int ans = 0;

        ArrayList<Character> ch = new ArrayList<>();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            ch.add(c);
        }
        for (int i = 0; i < ch.size() - 1; i++) {
            if (ch.get(i) == 'I' && (ch.get(i + 1) == 'V' || ch.get(i + 1) == 'X')) {
                if (ch.get(i + 1) == 'V') {
                    ans += 4;
                } else {
                    ans += 9;
                }
                ch.set(i, '0');
                ch.set(i + 1, '0');
            } else if (ch.get(i) == 'X' && (ch.get(i + 1) == 'L' || ch.get(i + 1) == 'C')) {
                if (ch.get(i + 1) == 'L') {
                    ans += 40;
                } else {
                    ans += 90;
                }
                ch.set(i, '0');
                ch.set(i + 1, '0');
            } else if (ch.get(i) == 'C' && (ch.get(i + 1) == 'D' || ch.get(i + 1) == 'M')) {
                if (ch.get(i + 1) == 'D') {
                    ans += 400;
                } else {
                    ans += 900;
                }
                ch.set(i, '0');
                ch.set(i + 1, '0');
            }
        }
        for (int i = 0; i < ch.size(); i++) {
            if (ch.get(i) == 'I') {
                ans += 1;
            } else if (ch.get(i) == 'V') {
                ans += 5;
            } else if (ch.get(i) == 'X') {
                ans += 10;
            } else if (ch.get(i) == 'L') {
                ans += 50;
            } else if (ch.get(i) == 'C') {
                ans += 100;
            } else if (ch.get(i) == 'D') {
                ans += 500;
            } else if (ch.get(i) == 'M') {
                ans += 1000;
            }
        }
        return ans;

    }
}
